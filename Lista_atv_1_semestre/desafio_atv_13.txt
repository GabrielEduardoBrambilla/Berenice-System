#include <stdio.h>
#include <stdlib.h>
#include <math.h>
int main()
{

  float h, massa, imc;

  printf("Digite sua altura: \n");
  scanf("%f", &h);
  printf("Digite sua massa: \n");
  scanf("%f", &massa);

  imc = massa / ( h * h);
  printf("Seu IMC e: %f \n", imc);
  printf("Voce esta: ");
  
  if(imc < 18.5){
    printf("Abaixo do peso");
  }
  if(imc >= 18.5 && imc <= 24.9){
    printf("Peso normal");
  }
  if(imc >= 25 && imc <= 29.9){
    printf("Sobrepeso");
  }
  if(imc >= 30 && imc <= 34.9){
    printf("Obesidade grau 1");
  } 
  if(imc >= 35 && imc <= 39.9){
    printf("Obesidade grau 2");
  }
  if(imc >= 40){
    printf("Obesidade morbida grau 3");
  }

  return 0;
}
